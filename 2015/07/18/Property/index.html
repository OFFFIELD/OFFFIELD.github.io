@Property

修饰符用法以及常见问题

修饰符

  • readwrite

    由系统创建setter&getter方法，默认值为readwrite，即property属性默认创建setter&getter方法

    @property (readwrite) NSObject *mObj;

  • readonly

    由系统自动创建getter方法，不创建setter方法

    @property (readonly) NSObject *mObj;

  • @synthesize

    Xcode4.5之后可以不写

    //在类成员变量中查找mObj和_mObj,若都没有，自动生成私有变量_mObj;
@synthesize mObj;

//如果想重新定义变量名
@synthesize  mObj = _mObjPrivate;

  • setter & getter

    重写setter或getter方法，覆盖readwrite或readonly所创建对应的方法

    //ViewController.h

#import <UIKit/UIKit.h>

@interface ViewController : UIViewController

@property (nonatomic, strong, getter = mGetNum, setter = mSetNum:) NSNumber *num;

@end

//ViewController.m

@interface ViewController ()
{
   //重新定义了setter和getter，@synthesize就会失效，需要重新定义私有变量
   NSNumber *_num;
}

@end

@implementation ViewController

- (void)viewDidLoad
{
  [super viewDidLoad];

  //只有用点语法才会调用setter方法
  self.num = @2;

  NSLog(@"%@",self.num);
}

//setter和getter里不可以用点语法，会导致死循环
- (void)mSetNum:(NSNumber *)num
{
      _num = num;
}

- (NSNumber *)mGetNum
{
   return _num;
}

@end
